package org.apache.commons.math4.geometry.euclidean.threed;

import java.text.NumberFormat;
import java.text.ParsePosition;
import java.util.Locale;

import org.apache.commons.math4.exception.MathParseException;
import org.apache.commons.math4.geometry.euclidean.threed.Cartesian3D;
import org.apache.commons.math4.geometry.euclidean.threed.Vector3DFormat;
import org.junit.Test;
import org.junit.Assert;
import org.junit.runner.RunWith;
import org.junit.experimental.theories.Theories;
import org.junit.experimental.theories.Theory;
import org.junit.experimental.theories.DataPoints;

public abstract @RunWith(Theories.class) class AutoPutTest {

    Vector3DFormat vector3DFormat = null;
    Vector3DFormat vector3DFormatSquare = null;

    protected abstract Locale getLocale();

    protected abstract char getDecimalCharacter();

    protected AutoPutTest() {
        vector3DFormat = Vector3DFormat.getInstance(getLocale());
        final NumberFormat nf = NumberFormat.getInstance(getLocale());
        nf.setMaximumFractionDigits(2);
        vector3DFormatSquare = new Vector3DFormat("[", "]", " : ", nf);
    }

    @Theory
	public void autoPutTest(Fixture fixture) {
        Cartesian3D c = new Cartesian3D(1, 1, 1);
        String expected = fixture._expected;
        String actual = vector3DFormat.format(c);
        Assert.assertEquals(expected, actual);
    }

    private static Object INPUT1 = null;

	private static String EXPECTED1 = "{1; 1; 1}";

	private static Object INPUT2 = null;

	private static String EXPECTED2 = "[1 : 1 : 1]";

	@DataPoints public static Fixture[] DATA={new Fixture(INPUT1,EXPECTED1),new Fixture(INPUT2,EXPECTED2)};

	public static class Fixture {
		private Object _input;
		private String _expected;

		public Fixture(Object _input, String _expected) {
			this._input = _input;
			this._expected = _expected;
		}
	}

}
