package org.apache.commons.math4.linear;

import java.text.NumberFormat;
import java.text.ParsePosition;
import java.util.Locale;

import org.junit.Test;
import org.junit.Assert;
import org.apache.commons.math4.exception.MathParseException;
import org.apache.commons.math4.linear.ArrayRealVector;
import org.apache.commons.math4.linear.RealVectorFormat;
import org.junit.runner.RunWith;
import org.junit.experimental.theories.Theories;
import org.junit.experimental.theories.Theory;
import org.junit.experimental.theories.DataPoints;

public abstract @RunWith(Theories.class) class AutoPutTest {

    RealVectorFormat realVectorFormat = null;
    RealVectorFormat realVectorFormatSquare = null;

    protected abstract Locale getLocale();

    protected abstract char getDecimalCharacter();

    public AutoPutTest() {
        realVectorFormat = RealVectorFormat.getInstance(getLocale());
        final NumberFormat nf = NumberFormat.getInstance(getLocale());
        nf.setMaximumFractionDigits(2);
        realVectorFormatSquare = new RealVectorFormat("[", "]", " : ", nf);
    }

    @Theory
	public void autoPutTest(Fixture fixture) {
        ArrayRealVector c = new ArrayRealVector(new double[] {fixture._input[0], fixture._input[1], fixture._input[2]});
        String expected =
            "{1"    + getDecimalCharacter() +
            fixture._expected[0] + getDecimalCharacter() +
            fixture._expected[1] + getDecimalCharacter() +
            fixture._expected[2];
        String actual = realVectorFormat.format(c);
        Assert.assertEquals(expected, actual);
    }

    private static double[] INPUT1 = { 1.23, 1.43, 1.63 };

	private static String[] EXPECTED1 = { "23; 1", "43; 1", "63}" };

	private static double[] INPUT2 = { 1.232323232323, 1.43434343434343, 1.633333333333 };

	private static String[] EXPECTED2 = { "2323232323; 1", "4343434343; 1", "6333333333}" };

	private static double[] INPUT3 = { -1.232323232323, 1.43, 1.63 };

	private static String[] EXPECTED3 = { "2323232323; 1", "43; 1", "63}" };

	private static double[] INPUT4 = { 1.23, -1.434343434343, 1.63 };

	private static String[] EXPECTED4 = { "23; -1", "4343434343; 1", "63}" };

	private static double[] INPUT5 = { 1.23, 1.43, -1.633333333333 };

	private static String[] EXPECTED5 = { "23; 1", "43; -1", "6333333333}" };

	@DataPoints public static Fixture[] DATA={new Fixture(INPUT1,EXPECTED1),new Fixture(INPUT2,EXPECTED2),new Fixture(INPUT3,EXPECTED3),new Fixture(INPUT4,EXPECTED4),new Fixture(INPUT5,EXPECTED5)};

	public static class Fixture {
		private double[] _input;
		private String[] _expected;

		public Fixture(double[] _input, String[] _expected) {
			this._input = _input;
			this._expected = _expected;
		}
	}
}
