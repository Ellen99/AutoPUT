public @RunWith(Parameterized.class) class ArcsSetTest {

    @Test
    public void testSideEmbedded() {

        ArcsSet s35 = new ArcsSet(3.0, 5.0, 1.0e-10);
        ArcsSet s16 = new ArcsSet(_input[0], 6.0, 1.0e-10);

        Assert.assertEquals(Side.BOTH,  s16.split(new Arc(3.0, 5.0, 1.0e-10)).getSide());
        Assert.assertEquals(Side.BOTH,  s16.split(new Arc(5.0, 3.0 + MathUtils.TWO_PI, 1.0e-10)).getSide());
        Assert.assertEquals(_input[1], s35.split(new Arc(_input[2], 6.0, 1.0e-10)).getSide());
        Assert.assertEquals(_input[3],  s35.split(new Arc(6.0, _input[4] + MathUtils.TWO_PI, 1.0e-10)).getSide());

    }

    @Parameters public static Collection<Object[]> data(){return Arrays.asList(new Object[][]{{{1.0,Side.MINUS,1.0,Side.PLUS,1.0},{}},{{4.0,Side.BOTH,4.0,Side.BOTH,4.0},{}}});}

	private Object _expected;
