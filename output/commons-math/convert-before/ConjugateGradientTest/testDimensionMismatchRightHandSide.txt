    public void testDimensionMismatchRightHandSide() {

        final IterativeLinearSolver solver;
        solver = new ConjugateGradient(10, 0., false);
        final ArrayRealVector b = new ArrayRealVector(_input[0]);
        final ArrayRealVector x = new ArrayRealVector(_input[1]);
        solver.solve(a, b, x);
    }

    @Parameters public static Collection<Object[]> data(){return Arrays.asList(new Object[][]{{{2,3},{}},{{3,2},{}}});}

	private double[] _input;
	private Object _expected;

		this._input = _input;
		this._expected = _expected;
	}
