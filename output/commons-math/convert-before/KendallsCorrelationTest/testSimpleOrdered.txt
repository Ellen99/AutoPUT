package org.apache.commons.math4.stat.correlation;

import java.util.Arrays;

import org.apache.commons.math4.TestUtils;

/**
 * Test cases for Kendall's Tau rank correlation.
 */
public @RunWith(Parameterized.class) class KendallsCorrelationTest extends PearsonsCorrelationTest {

    @Before
    public void setUp() {
        correlation = new KendallsCorrelation();
    }

    @Test
    public void testSimpleOrdered() {
        final int length = _input;
        final double[] xArray = new double[length];
        final double[] yArray = new double[length];
        for (int i = 0; i < length; i++) {
            xArray[i] = i;
            yArray[i] = i;
        }
        Assert.assertEquals(1.0, correlation.correlation(xArray, yArray), Double.MIN_VALUE);
    }

    @Parameters public static Collection<Object[]> data(){return Arrays.asList(new Object[][]{{10,{}},{16,{}}});}

	private double _input;
	private Object _expected;

	public AutoPUT(double _input, Object _expected) {
		this._input = _input;
		this._expected = _expected;
	}
}
