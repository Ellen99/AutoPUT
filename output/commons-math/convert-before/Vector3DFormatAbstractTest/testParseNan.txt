import org.apache.commons.math4.geometry.euclidean.threed.Vector3DFormat;
import org.junit.Test;
import org.junit.Assert;
import org.junit.runner.RunWith;

import org.junit.runners.Parameters;

public abstract @RunWith(Parameterized.class) class Vector3DFormatAbstractTest {

    protected abstract Locale getLocale();

    protected abstract char getDecimalCharacter();

    @Test
    public void tesNegativeInfinity() {
        Cartesian3D c = Cartesian3D.NEGATIVE_INFINITY;
        String expected = "{(-Infinity); (-Infinity); (-Infinity)}";
        String actual = vector3DFormat.format(c);
        Assert.assertEquals(expected, actual);
    }

    @Test
    public void testParseNan() throws MathParseException {
        String source = _input[0];
        Vector3D actual = vector3DFormat.parse(source);
        Assert.assertEquals(_input[1], actual);
    }

    @Parameters public static Collection<Object[]> data(){return Arrays.asList(new Object[][]{{{"{(NaN); (NaN); (NaN)}",Cartesian3D.NaN},{}},{{"{(Infinity); (Infinity); (Infinity)}",Cartesian3D.POSITIVE_INFINITY},{}},{{"{(-Infinity); (-Infinity); (-Infinity)}",Cartesian3D.NEGATIVE_INFINITY},{}}});}

	private Object[] _input;
	private Object _expected;

	public AutoPUT(Object[] _input, Object _expected) {
		this._input = _input;
		this._expected = _expected;
	}

}
