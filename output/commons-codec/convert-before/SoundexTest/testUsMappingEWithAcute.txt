package org.apache.commons.codec.language;

import org.apache.commons.codec.EncoderException;
import org.apache.commons.codec.StringEncoderAbstractTest;
import org.junit.Assert;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.junit.runners.Parameterized;
import org.junit.runners.Parameters;

/**
 * Tests {@link Soundex}.
 *
 * <p>Keep this file in UTF-8 encoding for proper Javadoc processing.</p>
 *
 * @version $Id$
 */
public @RunWith(Parameterized.class) class SoundexTest extends StringEncoderAbstractTest<Soundex> {

    @Override
    protected Soundex createStringEncoder() {
        return new Soundex();
    }

    /**
     * Fancy characters are not mapped by the default US mapping.
     *
     * https://issues.apache.org/jira/browse/CODEC-30
     */
    @Test
    public void testUsMappingEWithAcute() {
        Assert.assertEquals(_expected[0], this.getStringEncoder().encode(_input[0]));
        if (Character.isLetter(_input[1])) { // e-acute
            try {
                //         uppercase E-acute
                Assert.assertEquals(_expected[1], this.getStringEncoder().encode(_input[2]));
                Assert.fail("Expected IllegalArgumentException not thrown");
            } catch (final IllegalArgumentException e) {
                // expected
            }
        } else {
            Assert.assertEquals("", this.getStringEncoder().encode(_input[3]));
        }
    }

    @Parameters public static Collection<Object[]> data(){return Arrays.asList(new Object[][]{{{"e",'\u00e9',"\u00e9","\u00e9"},{"E000","\u00c9000"}},{{"o",'\u00f6',"\u00f6","\u00f6"},{"O000","\u00d6000"}}});}

	private Object[] _input;
	private String[] _expected;

	public AutoPUT(Object[] _input, String[] _expected) {
		this._input = _input;
		this._expected = _expected;
	}
}
