/**

public @RunWith(Parameterized.class) class QuotedPrintableCodecTest {

    private String constructString(final int [] unicodeChars) {
        final StringBuilder buffer = new StringBuilder();
        if (unicodeChars != null) {
            for (final int unicodeChar : unicodeChars) {
                buffer.append((char)unicodeChar);
            }
        }
        return buffer.toString();
    }

    @Test
    public void testBasicEncodeDecode() throws Exception {
        final QuotedPrintableCodec qpcodec = new QuotedPrintableCodec();
        final String plain = _expected[0];
        final String encoded = qpcodec.encode(plain);
        assertEquals(_input[0],
            _expected[1], encoded);
        assertEquals(_input[1],
            plain, qpcodec.decode(encoded));
    }

    @Parameters public static Collection<Object[]> data(){return Arrays.asList(new Object[][]{{{"Basic quoted-printable encoding test","Basic quoted-printable decoding test"},{"= Hello there =\r\n","=3D Hello there =3D=0D=0A"}},{{"Unsafe chars quoted-printable encoding test","Unsafe chars quoted-printable decoding test"},{"=\r\n","=3D=0D=0A"}}});}

	private String[] _input;

	private String[] _expected;

	public AutoPUT(String[] _input, String[] _expected) {
		this._input = _input;
		this._expected = _expected;
	}
}
