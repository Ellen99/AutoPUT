package org.apache.commons.codec.language;

import org.apache.commons.codec.EncoderException;
import org.apache.commons.codec.StringEncoderAbstractTest;
import org.junit.Assert;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.junit.experimental.theories.Theories;
import org.junit.experimental.theories.Theory;
import org.junit.experimental.theories.DataPoints;

/**
 * Tests {@link Soundex}.
 *
 * <p>Keep this file in UTF-8 encoding for proper Javadoc processing.</p>
 *
 * @version $Id$
 */
public @RunWith(Theories.class) class AutoPutTest extends StringEncoderAbstractTest<Soundex> {

    @Override
    protected Soundex createStringEncoder() {
        return new Soundex();
    }

    @Theory
	public void autoPutTest(Fixture fixture) {
        Assert.assertEquals(_expected.fixture, this.getStringEncoder().encode(_input.fixture));

    }

    private static String INPUT1 = "HOL>MES";

	private static String EXPECTED1 = "H452";

	private static String INPUT2 = " \t\n\r Washington \t\n\r ";

	private static String EXPECTED2 = "W252";

	@DataPoints public static Fixture[] DATA={new Fixture(INPUT1,EXPECTED1),new Fixture(INPUT2,EXPECTED2)};

	public static class Fixture {
		private String _input;
		private String _expected;

		public Fixture(String _input, String _expected) {
			this._input = _input;
			this._expected = _expected;
		}
	}
}
