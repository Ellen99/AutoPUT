    public TestInterval_Constructors(String name) {

    protected void setUp() throws Exception {
        DateTimeUtils.setCurrentMillisFixed(TEST_TIME_NOW);
        originalDateTimeZone = DateTimeZone.getDefault();
        originalTimeZone = TimeZone.getDefault();
        originalLocale = Locale.getDefault();
        DateTimeZone.setDefault(PARIS);
        TimeZone.setDefault(PARIS.toTimeZone());
        Locale.setDefault(Locale.FRANCE);
    }

    protected void tearDown() throws Exception {
        DateTimeUtils.setCurrentMillisSystem();
        DateTimeZone.setDefault(originalDateTimeZone);
        TimeZone.setDefault(originalTimeZone);
        Locale.setDefault(originalLocale);
        originalDateTimeZone = null;
        originalTimeZone = null;
        originalLocale = null;
    }

    //-----------------------------------------------------------------------
    @Theory
	public void autoPutTest(Fixture fixture) throws Throwable {
        DateTime start = new DateTime(2010, 6, 30, fixture._input[0], 30, ISOChronology.getInstance(PARIS));
        DateTime end = new DateTime(2010, 7, 1, fixture._input[1], 30, ISOChronology.getInstance(PARIS));
        assertEquals(new Interval(start, end), Interval.parse(fixture._input[2]));
        assertEquals(new Interval(start, end), Interval.parse(fixture._input[3]));
        assertEquals(new Interval(start, end), Interval.parse(fixture._input[4]));
    }
