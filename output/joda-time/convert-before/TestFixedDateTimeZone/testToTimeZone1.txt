package org.joda.time.tz;

import java.util.Date;
import java.util.TimeZone;

import junit.framework.TestCase;
import junit.framework.TestSuite;

import org.joda.time.DateTimeZone;
import org.junit.runner.RunWith;
import org.junit.runners.Parameterized;
import org.junit.runners.Parameters;

/**
 * Test cases for FixedDateTimeZone.
 *
 * @author Stephen Colebourne
 */
public @RunWith(Parameterized.class) class TestFixedDateTimeZone extends TestCase {

    public static void main(String[] args) {
        junit.textui.TestRunner.run(suite());
    }

    public static TestSuite suite() {
        return new TestSuite(TestFixedDateTimeZone.class);
    }

    protected void setUp() throws Exception {
        originalDateTimeZone = DateTimeZone.getDefault();
        DateTimeZone.setDefault(DateTimeZone.UTC);
    }

    protected void tearDown() throws Exception {
        DateTimeZone.setDefault(originalDateTimeZone);
    }

    public void testToTimeZone1() throws Exception {
        FixedDateTimeZone zone = new FixedDateTimeZone(_input[0], _input[1], _input[2], _input[3]);
        java.util.TimeZone tz = zone.toTimeZone();
        
        assertEquals(_expected[0], tz.getRawOffset());
        assertEquals(_expected[1], getOffset(tz, 1167638400000L));
        assertEquals(_expected[2], getOffset(tz, 1185951600000L));
    }

    /** Make test compile on JDK 1.3. */
    private int getOffset(TimeZone zone, long millis) {
        Date date = new Date(millis);
        if (zone.inDaylightTime(date)) {
            return zone.getRawOffset() + 3600000;
        }
        return zone.getRawOffset();
    }

	@Parameters public static Collection<Object[]> data(){return Arrays.asList(new Object[][]{{{"+00:01","+00:01",60000,60000},{60000,60000,60000}},{{"A","B",1,5},{1,1,1}}});}

	private Object[] _input;
	private double[] _expected;

	public AutoPUT(Object[] _input, double[] _expected) {
		this._input = _input;
		this._expected = _expected;
	}

}
