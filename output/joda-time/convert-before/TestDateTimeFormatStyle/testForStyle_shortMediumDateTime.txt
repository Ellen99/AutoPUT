package org.joda.time.format;

import java.text.DateFormat;
import java.util.Locale;
import java.util.SimpleTimeZone;
import java.util.TimeZone;

import junit.framework.TestCase;
import junit.framework.TestSuite;

import org.joda.time.DateTime;
import org.joda.time.DateTimeConstants;
import org.joda.time.DateTimeUtils;
import org.joda.time.DateTimeZone;
import org.junit.runner.RunWith;
import org.junit.runners.Parameterized;
import org.junit.runners.Parameters;

/**
 * This class is a Junit unit test for DateTimeFormat styles.
 *
 * @author Stephen Colebourne
 */
public @RunWith(Parameterized.class) class TestDateTimeFormatStyle extends TestCase {

    public static void main(String[] args) {
        junit.textui.TestRunner.run(suite());
    }

    public static TestSuite suite() {
        return new TestSuite(TestDateTimeFormatStyle.class);
    }

    protected void setUp() throws Exception {
        DateTimeUtils.setCurrentMillisFixed(TEST_TIME_NOW);
        originalDateTimeZone = DateTimeZone.getDefault();
        originalTimeZone = TimeZone.getDefault();
        originalLocale = Locale.getDefault();
        DateTimeZone.setDefault(LONDON);
        TimeZone.setDefault(TimeZone.getTimeZone("Europe/London"));
        Locale.setDefault(UK);
    }

    protected void tearDown() throws Exception {
        DateTimeUtils.setCurrentMillisSystem();
        DateTimeZone.setDefault(originalDateTimeZone);
        TimeZone.setDefault(originalTimeZone);
        Locale.setDefault(originalLocale);
        originalDateTimeZone = null;
        originalTimeZone = null;
        originalLocale = null;
    }

    //-----------------------------------------------------------------------
    public void testForStyle_shortMediumDateTime() throws Exception {
        DateTimeFormatter f = DateTimeFormat.forStyle(_input[0]);
        DateTime dt = new DateTime(2004, 6, 9, 10, 20, 30, 0);
        String expect = DateFormat.getDateTimeInstance(_expected[0], _expected[1], UK).format(dt.toDate());
        assertEquals(expect, f.print(dt));
        expect = DateFormat.getDateTimeInstance(_input[1], _input[2], US).format(dt.toDate());
        assertEquals(expect, f.withLocale(US).print(dt));
        expect = DateFormat.getDateTimeInstance(_input[3], _input[4], FRANCE).format(dt.toDate());
        assertEquals(expect, f.withLocale(FRANCE).print(dt));
    }

    @Parameters public static Collection<Object[]> data(){return Arrays.asList(new Object[][]{{{"SM",DateFormat.SHORT,DateFormat.MEDIUM,DateFormat.SHORT,DateFormat.MEDIUM},{DateFormat.SHORT,DateFormat.MEDIUM}},{{"SL",DateFormat.SHORT,DateFormat.LONG,DateFormat.SHORT,DateFormat.LONG},{DateFormat.SHORT,DateFormat.LONG}},{{"SF",DateFormat.SHORT,DateFormat.FULL,DateFormat.SHORT,DateFormat.FULL},{DateFormat.SHORT,DateFormat.FULL}},{{"MS",DateFormat.MEDIUM,DateFormat.SHORT,DateFormat.MEDIUM,DateFormat.SHORT},{DateFormat.MEDIUM,DateFormat.SHORT}},{{"ML",DateFormat.MEDIUM,DateFormat.LONG,DateFormat.MEDIUM,DateFormat.LONG},{DateFormat.MEDIUM,DateFormat.LONG}},{{"MF",DateFormat.MEDIUM,DateFormat.FULL,DateFormat.MEDIUM,DateFormat.FULL},{DateFormat.MEDIUM,DateFormat.FULL}}});}

	private Object[] _input;
	private double[] _expected;

	public AutoPUT(Object[] _input, double[] _expected) {
		this._input = _input;
		this._expected = _expected;
	}

}
