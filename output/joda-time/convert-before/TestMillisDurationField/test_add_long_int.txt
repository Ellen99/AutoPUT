/**
 * This class is a Junit unit test for PeriodFormatterBuilder.
 *
 * @author Stephen Colebourne
 */
public @RunWith(Parameterized.class) class TestMillisDurationField extends TestCase {

    public static void main(String[] args) {
        junit.textui.TestRunner.run(suite());
    }

    public static TestSuite suite() {
        return new TestSuite(TestMillisDurationField.class);
    }

    protected void setUp() throws Exception {
    }

    protected void tearDown() throws Exception {
    }

    //-----------------------------------------------------------------------
    public void test_add_long_int() {
        assertEquals(567L, MillisDurationField.INSTANCE.add(567L, _input[0]));
        assertEquals(567L + 1234L, MillisDurationField.INSTANCE.add(567L, _input[1]));
        assertEquals(567L - 1234L, MillisDurationField.INSTANCE.add(567L, _input[2]));
        try {
            MillisDurationField.INSTANCE.add(Long.MAX_VALUE, _input[3]);
            fail();
        } catch (ArithmeticException ex) {}
    }

    @Parameters public static Collection<Object[]> data(){return Arrays.asList(new Object[][]{{{0,1234,-1234,1},{}},{{0L,1234L,-1234L,1L},{}}});}

	private double[] _input;
	private Object _expected;

	public AutoPUT(double[] _input, Object _expected) {
		this._input = _input;
		this._expected = _expected;
	}

}
